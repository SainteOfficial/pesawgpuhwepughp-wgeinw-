-- RemoteEvent Monitor LocalScript mit Fehlerbehandlung
local erfolg, ergebnis = pcall(function()
    -- Versuche den RemoteEvent zu finden mit Timeout
    local startZeit = tick()
    local maxWartezeit = 10 -- Sekunden
    local remoteEvent
    
    print("Suche nach RemoteEvent...")
    
    repeat
        if game:GetService("ReplicatedStorage"):FindFirstChild("3aA") then
            if game:GetService("ReplicatedStorage"):FindFirstChild("3aA"):FindFirstChild("8c5664e6-c89f-45fb-9903-0208dfc1fe29") then
                remoteEvent = game:GetService("ReplicatedStorage"):FindFirstChild("3aA"):FindFirstChild("8c5664e6-c89f-45fb-9903-0208dfc1fe29")
                break
            end
        end
        wait(0.1)
    until (tick() - startZeit) > maxWartezeit
    
    if not remoteEvent then
        error("RemoteEvent konnte nicht gefunden werden nach " .. maxWartezeit .. " Sekunden")
    end
    
    print("RemoteEvent gefunden: " .. remoteEvent:GetFullName())
    local funktionsName = remoteEvent:GetFullName()
    
    -- Rest des Codes wie zuvor...
    -- Überwache eingehende Server-zu-Client Kommunikation
    remoteEvent.OnClientEvent:Connect(function(...)
        -- [Code wie im ursprünglichen Skript]
    end)
    
    -- Erstelle eine Kopie der originalen FireServer-Funktion
    local originalFireServer = remoteEvent.FireServer
    -- [Rest des Codes wie im ursprünglichen Skript]
end)

if not erfolg then
    warn("FEHLER beim Ausführen des RemoteEvent-Monitors: " .. tostring(ergebnis))
    
    -- Prüfe, ob der Pfad überhaupt existiert
    print("Überprüfe Pfad zu RemoteEvent:")
    if game:GetService("ReplicatedStorage") then
        print("- ReplicatedStorage existiert")
        if game:GetService("ReplicatedStorage"):FindFirstChild("3aA") then
            print("- '3aA' existiert")
            if game:GetService("ReplicatedStorage"):FindFirstChild("3aA"):FindFirstChild("8c5664e6-c89f-45fb-9903-0208dfc1fe29") then
                print("- '8c5664e6-c89f-45fb-9903-0208dfc1fe29' existiert")
            else
                print("- '8c5664e6-c89f-45fb-9903-0208dfc1fe29' existiert NICHT")
            end
        else
            print("- '3aA' existiert NICHT")
        end
    else
        print("- ReplicatedStorage existiert NICHT")
    end
end
